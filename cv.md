function toCamelCase(str){
  let camel = [];
  for (let i = 0; i < str.length; i++) {
    if (str[i] === '-' || str[i] ==='_') {
      i++;
      camel[i] = str[i].toUpperCase();
    } else {
      camel[i] = str[i];
    }
  }
  return camel.join('');
}
